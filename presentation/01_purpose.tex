    \tcbitem[

        boxrule=0.5mm
    ]
    \begin{itemize}[leftmargin=*, label=$\times$, nosep]
        \item \textbf{Manual Extraction:} Relies on ad-hoc queries and intuition.
        \item \textbf{Tribal Knowledge:} Criteria live in heads, not code.
        \item \textbf{High Noise:} Fails to distinguish signal from data drift.
        \item \textbf{Unscalable:} Testing debt grows superlinearly with complexity.
    \end{itemize}
    \vspace{2mm}
    \textit{\footnotesize \textbf{Consequence:} Regression testing becomes a bottleneck, forcing a choice between speed and safety.}
    

    % TAS Solution
    \tcbitem[
        colback=paper, 
        colframe=ink, 
        title={\bfseries \small THE TAS STANDARD (Deterministic \& Explanatory)},
        coltitle=paper,
        colbacktitle=ink, 
        fonttitle=\bfseries,
        sharp corners,
        boxrule=0.5mm
    ]
    \begin{itemize}[leftmargin=*, label=$\checkmark$, nosep]
        \item \textbf{Automated Validation:} Rules are codified and executable.
        \item \textbf{Living Documentation:} Tests explain the system's intent.
        \item \textbf{Precise Signal:} Zero-tolerance for unintended variance.
        \item \textbf{Invariant:} Correctness is established by explicit design.
    \end{itemize}
    \vspace{2mm}
    \textit{\footnotesize \textbf{Consequence:} Continuous evolution with mathematically guaranteed regression checks.}
